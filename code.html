        //function to clear the values using clear button.
        function clearValues(){
            document.loanForm.loan.value="";
            document.loanForm.term.value="";
            document.loanForm.interest.value="";
        
            document.getElementById("loanInfo").innerHTML="";
            document.getElementById("table").innerHTML="";
        }
        
        function verify(){
            var loan=document.loanForm.loan.value;
            var term =document.loanForm.term.value;
            var interest = document.loanForm.interest.value;
        
            if(loan <= 0 || isNaN(Number(loan)) ){
               document.loanForm.loan.value="";
        
            }
            else if(term <= 0 || parseInt(term) != term ){
                document.loanForm.term.value="";
            }
            else if(interest <= 0 || interest > 99 ||  isNaN(Number(interest)) ){
                document.loanForm.interest.value="";
            }
            else{
                calculate(parseFloat(loan), parseInt(term), parseFloat(interest));
            }
        }
        
        function calculate(loan, term, interest){
            i = interest/100;
            var monthly_payment = loan*(i/12)*Math.pow((1+i/12),term)/ (Math.pow((1+i/12),term)-1);
            monthly_payment =monthly_payment.toFixed(2);
        
           var total_interest = (monthly_payment*term)-loan;
           total_interest = total_interest.toFixed(2);
            
            document.getElementById("monthlyPayment").innerText =  monthly_payment;
            document.getElementById("totalPrincipal").innerText = loan;
            document.getElementById("totalInterest").innerText=total_interest;
            document.getElementById("totalCost").innerText=monthly_payment*term;
        
            let  remaining_balance = 0,
            interest_amount = 0,
            principal_payment = 0;
        
            for(i=1; i<=term;i++){
               remaining_balance= loan-monthly_payment;
              interest_amount= remaining_balance*interest/1200;
              principal_payment = monthly_payment - interest_amount;
              total_interest = interest_amount;
               remaining_balance = loan- principal_payment;
        
            document.getElementById("table").innerHTML += `${i}${monthly_payment}${principal_payment}${interest_amount}${total_interest}${remaining_balance}`
        
            }
        
        
        
        
        
        
        }
